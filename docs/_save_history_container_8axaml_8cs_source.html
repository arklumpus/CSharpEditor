60  public void Refresh()
61  {
62  this.FindControl<StackPanel>("FileContainer").Children.Clear();
63  Editor editor = this.FindAncestorOfType<Editor>();
64 
65  string autosaveDirectory = System.IO.Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData), Assembly.GetEntryAssembly().GetName().Name);
66  string savePath = System.IO.Path.Combine(autosaveDirectory, editor.Guid);
67 
68  string[] files;
69 
70  try
71  {
72  files = System.IO.Directory.GetFiles(savePath, "*.cs");
73  }
74  catch
75  {
76  files = new string[0];
77  }
78 
79 
80  long result = -1;
81 
82  List<(long, string)> sortedFiles = (from el in files let filename = System.IO.Path.GetFileNameWithoutExtension(el) where filename.StartsWith("autosave") || long.TryParse(filename, out result) let result2 = result orderby !filename.StartsWith("autosave") ? result2 : ((DateTimeOffset)new System.IO.FileInfo(el).LastWriteTimeUtc).ToUnixTimeSeconds() descending select !filename.StartsWith("autosave") ? (result2, el) : (((DateTimeOffset)new System.IO.FileInfo(el).LastWriteTimeUtc).ToUnixTimeSeconds(), el)).ToList();
83 
84  sortedFiles.Add((editor.OriginalTimeStamp, "original"));
85  sortedFiles.Sort((a, b) => Math.Sign(b.Item1 - a.Item1));
86 
87  int diffWidth = (int)Math.Max(10, Math.Floor(this.Bounds.Width - 32 - 150 - 32 - 32 - 22));
88 
89  string currentText = null;
90  int lineCount = -1;
91  Differ differ = new Differ();
92 
93  if (sortedFiles.Any())
94  {
95  currentText = editor.EditorControl.Text.ToString();
96  lineCount = editor.EditorControl.Text.Lines.Count;
97  }
98 
99  foreach ((long timestamp, string file) file in sortedFiles)
100  {
101  Grid itemGrid = new Grid();
102 
103  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(32, GridUnitType.Pixel));
104  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(1, GridUnitType.Pixel));
105  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(150, GridUnitType.Pixel));
106  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(1, GridUnitType.Pixel));
107  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(1, GridUnitType.Star));
108  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(1, GridUnitType.Pixel));
109  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(36, GridUnitType.Pixel));
110  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(1, GridUnitType.Pixel));
111  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(36, GridUnitType.Pixel));
112  itemGrid.ColumnDefinitions.Add(new ColumnDefinition(18, GridUnitType.Pixel));
113 
114  if (System.IO.Path.GetFileNameWithoutExtension(file.file).StartsWith("autosave"))
115  {
116  AutosaveIcon icon = new DiagnosticIcons.AutosaveIcon();
117  ToolTip.SetTip(icon, "Autosave");
118  itemGrid.Children.Add(icon);
119  }
120  else if (file.file == "original")
121  {
122  StartingIcon icon = new DiagnosticIcons.StartingIcon();
123  ToolTip.SetTip(icon, "Original file");
124  itemGrid.Children.Add(icon);
125  }
126  else
127  {
128  SaveIcon icon = new DiagnosticIcons.SaveIcon();
129  ToolTip.SetTip(icon, "Manual save");
130  itemGrid.Children.Add(icon);
131  }
132 
133  TextBlock timeBlock = new TextBlock() { Text = DateTimeOffset.FromUnixTimeSeconds(file.timestamp).DateTime.ToLocalTime().ToString("HH:mm:ss \\(dd MMM\\)", System.Globalization.CultureInfo.InvariantCulture), Margin = new Thickness(5, 2, 5, 2), VerticalAlignment = Avalonia.Layout.VerticalAlignment.Center };
134  Grid.SetColumn(timeBlock, 2);
135  itemGrid.Children.Add(timeBlock);
136 
137  DiffResult diffResult;
138 
139  if (file.file != "original")
140  {
141  diffResult = differ.CreateLineDiffs(currentText, System.IO.File.ReadAllText(file.file), false);
142  }
143  else
144  {
145  diffResult = differ.CreateLineDiffs(currentText, editor.OriginalText, false);
146  }
147 
148 
149  List<(int pos, int count, Colour colour)> diffLines = new List<(int pos, int count, Colour colour)>(from el in diffResult.DiffBlocks let pos = el.DeleteCountA > 0 ? el.DeleteStartA : el.InsertStartB orderby pos ascending select (pos, el.DeleteCountA > 0 ? el.DeleteCountA : el.InsertCountB, el.DeleteCountA > 0 ? RedColour : GreenColour));
150 
151  int currPos = 0;
152 
153  List<(double width, Colour colour)> diffBlocks = new List<(double width, Colour colour)>();
154 
155  for (int i = 0; i < diffLines.Count; i++)
156  {
157  if (diffLines[i].pos > currPos)
158  {
159  diffBlocks.Add((diffLines[i].pos - currPos, GreyColour));
160  currPos = diffLines[i].pos;
161  }
162 
163  diffBlocks.Add((diffLines[i].count, diffLines[i].colour));
164  currPos += diffLines[i].count;
165  }
166 
167  if (currPos < lineCount)
168  {
169  diffBlocks.Add((lineCount - currPos, GreyColour));
170  currPos = lineCount;
171  }
172 
173  Page diff = new Page(diffWidth, 20);
174  Graphics gpr = diff.Graphics;
175 
176  double currX = 0;
177 
178  for (int i = 0; i < diffBlocks.Count; i++)
179  {
180  double w = diffBlocks[i].width * diffWidth / currPos;
181  gpr.FillRectangle(currX, 0, w, 20, diffBlocks[i].colour);
182  currX += w;
183  }
184 
185  Viewbox diffBox = new Viewbox() { Child = diff.PaintToCanvas(false), VerticalAlignment = Avalonia.Layout.VerticalAlignment.Center, Height = 20, Stretch = Stretch.Fill };
186  Grid.SetColumn(diffBox, 4);
187  itemGrid.Children.Add(diffBox);
188 
189  if (file.file != "original")
190  {
191  Button restoreButton = new Button() { Content = new DiagnosticIcons.RestoreIcon(), Margin = new Thickness(3) };
192  Grid.SetColumn(restoreButton, 6);
193  ToolTip.SetTip(restoreButton, "Restore");
194  itemGrid.Children.Add(restoreButton);
195 
196  Button deleteButton = new Button() { Content = new DiagnosticIcons.DeleteIcon(), Margin = new Thickness(3) };
197  Grid.SetColumn(deleteButton, 8);
198  ToolTip.SetTip(deleteButton, "Delete");
199  itemGrid.Children.Add(deleteButton);
200 
201  restoreButton.Click += async (s, e) =>
202  {
203  try
204  {
205  string newText = System.IO.File.ReadAllText(file.file);
206  await editor.EditorControl.SetText(newText);
207 
208  if (editor.EditorControl.ShowLineChanges)
209  {
210  DiffResult diffResultFromLastSaved = editor.Differ.CreateLineDiffs(editor.PreSource + "\n" + editor.LastSavedText + "\n" + editor.PostSource, editor.PreSource + "\n" + newText + "\n" + editor.PostSource, false);
211  IEnumerable<int> changesFromLastSaved = (from el in diffResultFromLastSaved.DiffBlocks select Enumerable.Range(el.InsertStartB - editor.PreSourceText.Lines.Count, Math.Max(1, el.InsertCountB))).Aggregate(Enumerable.Empty<int>(), (a, b) => a.Concat(b));
212 
213  DiffResult diffResultFromOriginal = editor.Differ.CreateLineDiffs(editor.PreSource + "\n" + editor.OriginalText + "\n" + editor.PostSource, editor.PreSource + "\n" + newText + "\n" + editor.PostSource, false);
214  IEnumerable<int> changesFromOriginal = (from el in diffResultFromOriginal.DiffBlocks select Enumerable.Range(el.InsertStartB - editor.PreSourceText.Lines.Count, Math.Max(1, el.InsertCountB))).Aggregate(Enumerable.Empty<int>(), (a, b) => a.Concat(b));
215 
216  editor.SetLineDiff(changesFromLastSaved, changesFromOriginal);
217  }
218 
219  }
220  catch { }
221 
222  Refresh();
223  };

236  else
237  {
238  Button restoreButton = new Button() { Content = new DiagnosticIcons.RestoreIcon(), Margin = new Thickness(3) };
239  Grid.SetColumn(restoreButton, 6);
240  ToolTip.SetTip(restoreButton, "Restore");
241  itemGrid.Children.Add(restoreButton);

244  {
245  string newText = editor.OriginalText;
246  await editor.EditorControl.SetText(newText);
247 
248  if (editor.EditorControl.ShowLineChanges)
249  {
250  DiffResult diffResultFromLastSaved = editor.Differ.CreateLineDiffs(editor.PreSource + "\n" + editor.LastSavedText + "\n" + editor.PostSource, editor.PreSource + "\n" + newText + "\n" + editor.PostSource, false);
251  IEnumerable<int> changesFromLastSaved = (from el in diffResultFromLastSaved.DiffBlocks select Enumerable.Range(el.InsertStartB - editor.PreSourceText.Lines.Count, Math.Max(1, el.InsertCountB))).Aggregate(Enumerable.Empty<int>(), (a, b) => a.Concat(b));
252 
253  DiffResult diffResultFromOriginal = editor.Differ.CreateLineDiffs(editor.PreSource + "\n" + editor.OriginalText + "\n" + editor.PostSource, editor.PreSource + "\n" + newText + "\n" + editor.PostSource, false);
254  IEnumerable<int> changesFromOriginal = (from el in diffResultFromOriginal.DiffBlocks select Enumerable.Range(el.InsertStartB - editor.PreSourceText.Lines.Count, Math.Max(1, el.InsertCountB))).Aggregate(Enumerable.Empty<int>(), (a, b) => a.Concat(b));
255 
256  editor.SetLineDiff(changesFromLastSaved, changesFromOriginal);
257  }
258 
259  Refresh();
260  };

263  this.FindControl<StackPanel>("FileContainer").Children.Add(itemGrid);
