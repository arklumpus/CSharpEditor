19 using Avalonia.Controls;
20 using Avalonia.Media;
21 using Microsoft.CodeAnalysis.Text;
22 using System.Collections.Generic;
23 using System.Linq;
24 
25 namespace CSharpEditor
26 {
27  internal class CSharpSourceEditorScrollBarMarker : Control
28  {
29  private CSharpSourceEditor Editor;
30 
31  public CSharpSourceEditorScrollBarMarker(CSharpSourceEditor editor)
32  {
33  this.Editor = editor;
34  this.Width = 16;
35  this.HorizontalAlignment = Avalonia.Layout.HorizontalAlignment.Right;
36  this.IsHitTestVisible = false;
37  }
38 
39  private static IBrush CaretPositionBrush = new SolidColorBrush(Color.FromRgb(0, 0, 205));
40  private static SolidColorBrush BreakpointBrush = new SolidColorBrush(Color.FromRgb(171, 97, 107));
41  private static SolidColorBrush SearchBrush = new SolidColorBrush(Color.FromRgb(246, 185, 127));
42 
43  public override void Render(DrawingContext context)
44  {
45  base.Render(context);
46 
47  if (Editor.ShowScrollbarOverview && Editor.Extent.Height > Editor.Bounds.Height)
48  {
49  double totalHeight = this.Bounds.Height - 36;
50 
51  if (Editor.Extent.Width > Editor.Bounds.Width)
52  {
53  totalHeight -= 16;
54  }
55 
56  double totalLines = Editor.Text.Lines.Count;
57 
58  foreach (HighlightedLineRange range in this.Editor.HighlightedLines)
59  {
60  context.FillRectangle(range.HighlightBrush, CSharpSourceEditorCaret.Round(new Avalonia.Rect(0, 16 + range.LineSpan.Start / totalLines * totalHeight, 4, (range.LineSpan.Length + 1) / totalLines * totalHeight), new Avalonia.Size(1, 1)));
61  }
62 
63  HashSet<int> redLines = new HashSet<int>();
64  HashSet<int> greenLines = new HashSet<int>();
65 
66  for (int i = 0; i < this.Editor.Markers.Count; i++)
67  {
68  LinePositionSpan markerSpan = this.Editor.Text.Lines.GetLinePositionSpan(this.Editor.Markers[i].Span);
69  if (this.Editor.Markers[i].MarkerPen == ErrorContainer.WarningPen)
70  {
71  foreach (int j in Enumerable.Range(markerSpan.Start.Line, markerSpan.End.Line - markerSpan.Start.Line + 1))
72  {
73  greenLines.Add(j);
74  }
75  }
76  else
77  {
78  foreach (int j in Enumerable.Range(markerSpan.Start.Line, markerSpan.End.Line - markerSpan.Start.Line + 1))
79  {
80  redLines.Add(j);
81  }
82  }
83  }
84 
85 
86  List<int> sortedRedLines = new List<int>(redLines);
87  List<int> sortedGreenLines = new List<int>(greenLines);
88 
89  sortedRedLines.Sort();
90  sortedGreenLines.Sort();
91 
92  IEnumerable<HighlightedLineRange> greenLineSpans = from el in (from el in sortedGreenLines where !sortedRedLines.Contains(el) select new TextSpan(el, 0)).Join() select new HighlightedLineRange(el, Brushes.Green);
93  IEnumerable<HighlightedLineRange> redLineSpans = from el in (from el in sortedRedLines select new TextSpan(el, 0)).Join() select new HighlightedLineRange(el, Brushes.Red);
94 
95  foreach (HighlightedLineRange range in greenLineSpans.Concat(redLineSpans))
96  {
97  context.FillRectangle(range.HighlightBrush, CSharpSourceEditorCaret.Round(new Avalonia.Rect(12, 16 + range.LineSpan.Start / totalLines * totalHeight, 4, (range.LineSpan.Length + 1) / totalLines * totalHeight), new Avalonia.Size(1, 1)));
98  }
99 
100  if (this.Editor.SearchReplace.IsVisible)
101  {
102  HashSet<int> searchMatches = new HashSet<int>();
103 
104  foreach (SearchSpan span in this.Editor.SearchSpans)
105  {
106  LinePositionSpan lineSpan = this.Editor.Text.Lines.GetLinePositionSpan(span.Span);
107  foreach (int j in Enumerable.Range(lineSpan.Start.Line, lineSpan.End.Line - lineSpan.Start.Line + 1))
108  {
109  searchMatches.Add(j);
110  }
111  }
112 
113  foreach (int line in searchMatches)
114  {
115  context.FillRectangle(SearchBrush, CSharpSourceEditorCaret.Round(new Avalonia.Rect(6, 16 + line / totalLines * totalHeight, 4, 1.0 / totalLines * totalHeight), new Avalonia.Size(1, 1)));
116  }
117  }
118 
119  foreach (int offset in this.Editor.Text.ToString().AllIndicesOf(Utils.BreakpointMarker))
120  {
121  int line = this.Editor.Text.Lines.GetLinePosition(offset).Line;
122 
123  context.FillRectangle(BreakpointBrush, CSharpSourceEditorCaret.Round(new Avalonia.Rect(6, 16 + line / totalLines * totalHeight, 4, 1.0 / totalLines * totalHeight), new Avalonia.Size(1, 1)));
124  }
125 
126  context.FillRectangle(CaretPositionBrush, CSharpSourceEditorCaret.Round(new Avalonia.Rect(0, 16 + (Editor.Text.Lines.GetLinePosition(Editor.CaretOffset).Line + 0.5) / totalLines * totalHeight - 1, 16, 2), new Avalonia.Size(1, 1)));
127  }
128  }
129  }
130 }
